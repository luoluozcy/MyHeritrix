//获的月的天数
Calendar.monthDayNum = function(year, month){
    var tdate = new Date(year, month + 1, 0);
    return tdate.getDate();
}
//获的指定1号是星期几
Calendar.dayInWeek = function(year, month, date){
    var tdate = new Date(year, month, date);
    return tdate.getDay();
}
//打印日历
Calendar.draw = function(callback){
    var year = Calendar.year;
    var month = Calendar.month;
    var currentDay = Calendar.date;
    var firstDayInWeek = Calendar.dayInWeek(year, month, 1);
    var monthDayNum = Calendar.monthDayNum(year, month, 0);
    callback(year, month, currentDay, firstDayInWeek, monthDayNum);
}
//上一月 下一月
Calendar.NP = function(yearID, monthID, state, fun, loadData){
    var yDom = $(yearID);
    var mDom = $(monthID);
    var year = parseInt(yDom.innerHTML);
    var month = parseInt(mDom.innerHTML);
    if (state == "next") {
        month = month + 1;
        if (month > 12) {
            month = 1;
            year = year + 1;
        }
    }
    if (state == "previous") {
        month = month - 1;
        if (month == 0) {
            month = 12;
            year = year - 1;
        }
    }
    Calendar.month = month - 1;
    Calendar.year = year;
    yDom.innerHTML = year;
    mDom.innerHTML = month;
    if (loadData) {
        var tm;
        if (month < 10) {
            tm = "0" + month.toString();
        }
        else {
            tm = month;
        }
        NTES.ajax.send("http://travel.163.com/datemonth/" + year + "/" + tm, "get", null, {
            onSuccess: getLoadData
        })
    }
    Calendar.draw(fun)
}
//显示日历
Calendar.print = function(year, month, currentDay, firstDayInWeek, monthDayNum){
    var year = year;
    var month = month;
    var currentDay = currentDay;
    var firstDayInWeek = firstDayInWeek;
    var monthDayNum = monthDayNum;
    var displayMonth, displayDay;
    var liarray = [];
    if (month < 9) {
        month = month + 1
        displayMonth = "0" + month;
    }
    else {
        displayMonth = month + 1;
    }
    for (var i = 0; i < firstDayInWeek; i++) {
        liarray.push("<li></li>");
    }
    for (var j = 1; j <= monthDayNum; j++) {
        if (j < 10) {
            displayDay = "0" + j;
        }
        else {
            displayDay = j;
        }
        if (j == currentDay) {
            liarray.push("<li><a class=\"current\" href=\"http://snapshot.news.163.com/wgethtml/http+!!travel.163.com!/" + year + "-" + displayMonth + "/" + displayDay + "/12.html\">" + j + "</a></li>");
        }
        else {
            liarray.push("<li><a href=\"http://snapshot.news.163.com/wgethtml/http+!!travel.163.com!/" + year + "-" + displayMonth + "/" + displayDay + "/12.html\">" + j + "</a></li>");
        }
    }
    var str = liarray.join("");
    $("#calendarList").innerHTML = str;
}
